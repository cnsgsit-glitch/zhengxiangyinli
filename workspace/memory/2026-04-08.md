def crop_simple(src_path, target_w=300, target_h=300):

    # 从顶部5%处开始取目标高度
    crop_top = int(new_h * 0.05)
    crop_bottom = crop_top + target_h

    crop_left = (new_w - target_w) // 2
    return img.crop((crop_left, crop_top, crop_left + target_w, crop_bottom))
